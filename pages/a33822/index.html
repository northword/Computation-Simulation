linux编译gnuplot

# 使用默认GCC编译gnuplot

无需，学校已安装。

cd /tmp
wget https://jaist.dl.sourceforge.net/project/gnuplot/gnuplot/5.2.5/gnuplot-5.2.5.tar.gz
tar xzvf gnuplot-5.2.5.tar.gz 
cd gnuplot-5.2.5/
./configure --prefix=/public/home/zjb/app/gunplot/
make & make install
